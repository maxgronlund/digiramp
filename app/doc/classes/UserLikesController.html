def index
  if params[:commit] == 'Go'
    @whipe_users = true
    params.delete :commit
    session[:user_query] = params[:query]
  end
  
  session[:user_query] = nil if params[:clear] == 'clear'
  params[:query]  = session[:user_query]
  @users = @user.liked_users.search(params[:query]).page(params[:page]).per(8)
 
end

def new
