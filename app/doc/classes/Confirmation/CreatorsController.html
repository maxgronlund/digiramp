def edit
  begin
    @common_work_ipi  = CommonWorkIpi.find_by(uuid: params[:uuid]) 
    @user             = User.find_by_password_reset_token!(params[:id])
    @common_work      = @common_work_ipi.common_work
    @invitor          = @common_work.account.user
  rescue
    not_found
  end
end
